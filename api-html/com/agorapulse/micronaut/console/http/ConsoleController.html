18 package com.agorapulse.micronaut.console.http;
19 
20 import com.agorapulse.micronaut.console.ConsoleException;
21 import com.agorapulse.micronaut.console.ConsoleSecurityException;
22 import com.agorapulse.micronaut.console.ConsoleService;
23 import com.agorapulse.micronaut.console.ExecutionResult;
24 import com.agorapulse.micronaut.console.Script;
25 import com.agorapulse.micronaut.console.User;
26 import com.agorapulse.micronaut.console.ide.DslGenerator;
27 import com.agorapulse.micronaut.console.util.ExceptionSanitizer;
28 import io.micronaut.http.HttpResponse;
29 import io.micronaut.http.MediaType;
30 import io.micronaut.http.annotation.Body;
31 import io.micronaut.http.annotation.Consumes;
32 import io.micronaut.http.annotation.Controller;
33 import io.micronaut.http.annotation.Error;
34 import io.micronaut.http.annotation.Get;
35 import io.micronaut.http.annotation.Header;
36 import io.micronaut.http.annotation.PathVariable;
37 import io.micronaut.http.annotation.Post;
38 import io.micronaut.http.annotation.Produces;
39 import io.micronaut.http.hateoas.JsonError;
40 
41 import javax.annotation.Nullable;
42 import java.util.List;
43 import java.util.Map;
44 import java.util.function.Function;
45 import java.util.stream.Collectors;
46 
47 @Controller("${console.path:/console}")
48 public class ConsoleController {
49 
50     private final ConsoleService service;
51     private final ExceptionSanitizer sanitizer;
52     private final Map<String, DslGenerator> generators;
53 
54     public ConsoleController(ConsoleService service, ExceptionSanitizer sanitizer, List<DslGenerator> generatorList) {
55         this.service = service;
56         this.sanitizer = sanitizer;
57         this.generators = generatorList.stream().collect(Collectors.toMap(DslGenerator::getScriptType, Function.identity()));
58     }
59 
60     @Post("/execute")
61     @Consumes(MediaType.ALL)
62     @Produces(MediaType.APPLICATION_JSON)
63     public ExecutionResult execute(@Body String body, @Nullable @Header("Content-Type") String contentType, User user) {
64         return service.execute(new Script(service.getLanguageForMimeType(contentType), body, user));
65     }
66 
67     @Post("/execute/result")
68     @Consumes(MediaType.ALL)
69     @Produces(MediaType.TEXT_PLAIN)
70     public HttpResponse<String> executeAndReturnText(@Body String body, @Nullable @Header("Content-Type") String contentType, User user) {
71         try {
72             return HttpResponse.ok(service.execute(new Script(service.getLanguageForMimeType(contentType), body, user)).toString());
73         } catch (ConsoleException e) {
74             return HttpResponse.badRequest(sanitizer.extractMessage(e) + "\n\n" + e.getScript());
75         }
76     }
77 
78     @Get("/dsl/{type}")
79     @Produces("text/plain")
80     public HttpResponse<String> generateDslFile(@PathVariable("type") String type) {
81         DslGenerator generator = generators.get(type);
82 
83         if (generator == null) {
84             return HttpResponse.notFound("Generator for " + type + " does not exist! Available " + String.join(", ", generators.keySet()));
85         }
86 
87         return HttpResponse.ok(generator.generateScript());
88     }
89 
90     @Error(ConsoleException.class)
91     public HttpResponse<JsonError> consoleException(ConsoleException exception) {
92         ScriptJsonError error = new ScriptJsonError(exception.getScript(), sanitizer.extractMessage(exception));
93         if (exception instanceof ConsoleSecurityException) {
94             return HttpResponse.unauthorized();
95         }
96         return HttpResponse.badRequest(error);
97     }
