18  package com.agorapulse.micronaut.console;
19  
20  import java.util.Collections;
21  import java.util.Map;
22  
23  /**
24   * Provides additional binding values for the script.
25   */
26  public interface BindingProvider {
27  
28      /**
29       * @return the additional bindings for the script
30       */
31      Map<String, ?> getBinding();
32  
33      static String purifyClassName(String className) {
34          if (!className.contains("$")) {
35              return className;
36          }
37  
38          if (className.contains(".$") && className.contains("Definition$Intercepted")) {
39              return className
40                  // Micronaut 3.x
41                  .replace("$Definition$Intercepted", "")
42                  // Micronaut 1.x and 2.x
43                  .replace("Definition$Intercepted", "")
44                  .replace(".$", ".");
45          }
46  
47          return className
48              .replace("$Intercepted", "")
49              .replace(".$", ".")
50              .replace('$', '.');
51      }
52  
53      static Map<String, ?> getDefaultBindingsStubs() {
54          return Collections.singletonMap("user", User.anonymous());
55      }
56  
57  }
