18  package com.agorapulse.micronaut.console.groovy;
19  
20  import com.agorapulse.micronaut.console.ConsoleEngine;
21  import com.agorapulse.micronaut.console.ExecutionResult;
22  import groovy.lang.Binding;
23  import groovy.lang.GroovyShell;
24  import io.micronaut.context.annotation.Requires;
25  import org.codehaus.groovy.control.CompilerConfiguration;
26  
27  import javax.inject.Singleton;
28  import java.io.PrintWriter;
29  import java.io.StringWriter;
30  import java.util.Arrays;
31  import java.util.LinkedHashMap;
32  import java.util.List;
33  import java.util.Map;
34  
35  @Singleton
36  @Requires(classes = GroovyShell.class)
37  public class GroovyConsoleEngine implements ConsoleEngine {
38  
39      private static final String LANGUAGE = "groovy";
40      private static final List<String> MIME_TYPES = Arrays.asList("application/groovy", "text/groovy");
41  
42      private final List<CompilerConfigurationCustomizer> customizers;
43  
44      public GroovyConsoleEngine(List<CompilerConfigurationCustomizer> customizers) {
45          this.customizers = customizers;
46      }
47  
48      @Override
49      public ExecutionResult execute(String code, Map<String, Object> bindings) {
50          StringWriter sw = new StringWriter();
51          PrintWriter writer = new PrintWriter(sw);
52  
53          Map<String, Object> groovyBindings = new LinkedHashMap<>(bindings);
54          groovyBindings.put("out", writer);
55  
56          CompilerConfiguration configuration = new CompilerConfiguration();
57          customizers.forEach(c -> c.customize(configuration));
58  
59          GroovyShell groovyShell = new GroovyShell(new Binding(groovyBindings), configuration);
60  
61          Object result = groovyShell.evaluate(code);
62  
63          return new ExecutionResult(result, sw.toString());
64      }
65  
66      @Override
67      public String getLanguage() {
68          return LANGUAGE;
69      }
70  
71      @Override
72      public List<String> getSupportedMimeTypes() {
73          return MIME_TYPES;
74      }
75  
76  }
